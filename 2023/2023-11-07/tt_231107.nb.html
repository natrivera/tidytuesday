house_dat <- house |>

house_dist_agg <- house_dat |>
  mutate(runoff = ifelse(is.na(runoff) , FALSE , runoff)) |>
  filter(stage == 'GEN' , runoff != TRUE , special != TRUE) |>
  group_by(year , state, state_po , dist) |>
  summarise(votes = max(totalvotes) ) |>
